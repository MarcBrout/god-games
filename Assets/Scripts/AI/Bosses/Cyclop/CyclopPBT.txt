tree "Root"
	fallback
		IsDead
		sequence
			tree("__AcquireNewTargetIfNeeded__")
			tree("__ImproveCyclopAttacksIfNeeded__")
			fallback
				tree("__RainOfRocks__")
				tree("__EvolvedAttack__")
				tree("__BasicAttack__")
				tree("__idle__")

tree "__AcquireNewTargetIfNeeded__"
	fallback
		sequence
			fallback
				not HasCurrentTarget
				HasToSwitchTarget
			AcquireNewTarget
		Success
tree "__ImproveCyclopAttacksIfNeeded__"
	fallback
		sequence
			DoesARopeHasBeenCutted
			ActivateRainOfRocksState
			ImproveBasicAttackLevel
		Success

tree "__BasicAttack__"
	sequence
		HasCurrentTarget
		CanUseBasicAttack
		UseBasicAttack

tree "__EvolvedAttack__"
	sequence
		HasCurrentTarget
		IsEvolvedAttackAvailable
		UseEvolvedAttack

tree "__RainOfRocks__"
	sequence
		IsRainOfRocksAvailable
		ActivateRainOfRocks
		WaitForRainOfRocksDuration
		DeactivateRainOfRocks

tree "__idle__"
	Idle
